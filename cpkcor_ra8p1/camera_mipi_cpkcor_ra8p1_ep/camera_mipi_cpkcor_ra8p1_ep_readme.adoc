=== 驱动文件

对相机的驱动文件都在 src/camera 文件夹中，使用方式如下

[source,c,subs="verbatim,attributes"]
----
/* 相机控制结构体 */
static Camera s_camera_ov5640_mipi;

void hal_entry(void)
{
    /* 相机配置结构体 */
    CameraConfig ov5640_config_mipi;

    /* 相机使用 IIC 接口进行配置，因此需要先初始化 IIC */
    IIC_Init();

    /* 填写相机配置 */
    ov5640_config_mipi.model = CAMERA_MODEL_OV5640;
    ov5640_config_mipi.interface = CAMERA_INTERFACE_MIPI_CSI;
    ov5640_config_mipi.length = 1024;
    ov5640_config_mipi.width = 600;
    ov5640_config_mipi.peripheral_inst = &g_vin_ov5640;
    /* 相机初始化 */
    CameraInit(&s_camera_ov5640_mipi, &ov5640_config_mipi);
    /* 开始捕获相机的图像 */
    CameraStartCapture(&s_camera_ov5640_mipi, NULL);

    ...
    /* 检查是否获取到了完整的一帧图像 */
    ret = CameraIsCaptureDone(&s_camera_ov5640_mipi);
    /* 如果已有一帧完整的图像，那么相机控制结构体下的变量 image_ready 会指向
     * 对应的图像接收块，随后可根据应用逻辑进行处理。 */
    if (s_camera_ov5640_mipi.image_ready != NULL) {
        ...
    }
    ...
}
----
